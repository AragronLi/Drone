CODE SEGMENT
    ASSUME CS:CODE

START:
    MOV DX, 283H
    MOV AL, 80H
    OUT DX, AL

    MOV CX, 1F4H

ABC:
    MOV DX, 281H
    MOV AL, 03H
    OUT DX, AL

    MOV DX, 280H
    MOV AL, 76H
    OUT DX, AL

    MOV DX, 281H
    MOV AL, 01H
    OUT DX, AL

    CALL DELAY

    MOV DX, 281H
    MOV AL, 03H
    OUT DX, AL

    MOV DX, 280H
    MOV AL, 73H
    OUT DX, AL

    MOV DX, 281H
    MOV AL, 02H
    OUT DX, AL

    CALL DELAY

    LOOP ABC

    MOV DX, 281H
    MOV AL, 03H
    OUT DX, AL

    MOV AH, 4CH
    INT 21H

DELAY PROC
    PUSH BX
    PUSH CX
    MOV BX, 5
LP1:
    MOV CX, 0FFFFH
LP2:
    LOOP LP2
    DEC BX
    JNZ LP1
    POP CX
    POP BX
    RET
DELAY ENDP

CODE ENDS
    END START
