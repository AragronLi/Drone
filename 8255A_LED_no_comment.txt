CODE SEGMENT
    ASSUME CS:CODE, DS:CODE
    ORG 100H

START:
    JMP INIT

COUNT   DB 0

INIT:
    MOV DX, 283H
    MOV AL, 82H
    OUT DX, AL
    MOV BL, 0

MAIN_LOOP:
    MOV AL, BL
    MOV DX, 280H
    OUT DX, AL
    CALL DELAY_1S
    MOV DX, 281H
    IN  AL, DX
    AND AL, 01H
    CMP AL, 01H
    JE  LED_OFF
    INC BL
    AND BL, 0FH
    JMP MAIN_LOOP

LED_OFF:
    MOV DX, 280H
    MOV AL, 00H
    OUT DX, AL

WAIT_LOW:
    MOV DX, 281H
    IN  AL, DX
    AND AL, 01H
    CMP AL, 00H
    JE  EXIT_PROG
    JMP WAIT_LOW

EXIT_PROG:
    MOV AH, 4CH
    INT 21H

DELAY_1S PROC
    PUSH CX
    PUSH DX
    MOV  CX, 5F6H

DELAY_OUTER:
    MOV  DX, 0FFFFH

DELAY_INNER:
    DEC  DX
    JNZ  DELAY_INNER
    DEC  CX
    JNZ  DELAY_OUTER
    POP  DX
    POP  CX
    RET
DELAY_1S ENDP

CODE ENDS
    END START
