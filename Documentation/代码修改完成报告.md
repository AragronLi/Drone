## 📝 后续工作

### 需要在main.c中添加

```c
#include "oled_ssd1306.h"
#include "sensor_compass.h"

int main(void) {
    // 系统初始化
    SystemInit();
    Delay_Init();
    USART_Printf_Init(115200);
    
    // I2C3设备初始化
    oled_init();      // OLED旧方案
    compass_init();   // 罗盘旧方案
    
    // 测试显示
    oled_show_string(0, 0, "CH32H417", 8);
    oled_show_string(0, 16, "I2C3 OK!", 8);
    oled_refresh();
    
    // 主循环
    while(1) {
        // 读取罗盘数据
        compass_data_t compass;
        if (compass_read_data(&compass)) {
            float heading = compass_get_heading(
                compass.mag_x, 
                compass.mag_y, 
                0.0f  // 磁偏角
            );
            printf("Heading: %.1f°\r\n", heading);
        }
        
        Delay_Ms(100);
    }
}
```
