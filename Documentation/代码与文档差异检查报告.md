```c
/********************************** (C) COPYRIGHT  *******************************
 * 文件名              : sensor_compass.h

 *                       I2C3: PA14(SCL) + PA13(SDA), 400kHz

#define __SENSOR_COMPASS_H

#include "ch32h417.h"

/* ---- 罗盘旧方案 I2C地址 ---- */
#define 罗盘旧方案_ADDR       0x0D

/* ---- 罗盘旧方案 寄存器 ---- */
#define 罗盘旧方案_REG_DATA_X_L   0x00
#define 罗盘旧方案_REG_DATA_X_H   0x01
#define 罗盘旧方案_REG_DATA_Y_L   0x02
#define 罗盘旧方案_REG_DATA_Y_H   0x03
#define 罗盘旧方案_REG_DATA_Z_L   0x04

#define 罗盘旧方案_REG_TEMP_H     0x08
#define 罗盘旧方案_REG_CONTROL1   0x09
#define 罗盘旧方案_REG_CONTROL2   0x0A

    int16_t mag_z;
```

**新建文件**: `V3F/Driver/sensor_compass.c`

```c
/********************************** (C) COPYRIGHT  *******************************
 * 文件名              : sensor_compass.c
 * 描述                : 罗盘旧方案传感器 I2C 驱动实现
 *                       I2C3: PA14(SCL) + PA13(SDA), 400kHz
 *                       与OLED OLED旧方案共享I2C3总线
 *******************************************************************************/
#include "sensor_compass.h"
#include "debug.h"
#include <math.h>

#define COMPASS_I2C     I2C3

/* ---- I2C写寄存器 ---- */
static void compass_write_reg(uint8_t reg, uint8_t val)
{
    // 等待总线空闲
    while (I2C_GetFlagStatus(COMPASS_I2C, I2C_FLAG_BUSY));
    
    // 发送起始信号
    I2C_GenerateSTART(COMPASS_I2C, ENABLE);
    while (!I2C_CheckEvent(COMPASS_I2C, I2C_EVENT_MASTER_MODE_SELECT));
    
    // 发送设备地址 (写)
    I2C_Send7bitAddress(COMPASS_I2C, 罗盘旧方案_ADDR << 1, I2C_Direction_Transmitter);
    while (!I2C_CheckEvent(COMPASS_I2C, I2C_EVENT_MASTER_TRANSMITTER_MODE_SELECTED));
    
    // 发送寄存器地址
    I2C_SendData(COMPASS_I2C, reg);
    while (!I2C_CheckEvent(COMPASS_I2C, I2C_EVENT_MASTER_BYTE_TRANSMITTED));
    
    // 发送数据
    I2C_SendData(COMPASS_I2C, val);
    while (!I2C_CheckEvent(COMPASS_I2C, I2C_EVENT_MASTER_BYTE_TRANSMITTED));
    
    // 发送停止信号
    I2C_GenerateSTOP(COMPASS_I2C, ENABLE);
}

/* ---- I2C读寄存器 ---- */
static uint8_t compass_read_reg(uint8_t reg)
{
    uint8_t val;
    
    // 等待总线空闲
    while (I2C_GetFlagStatus(COMPASS_I2C, I2C_FLAG_BUSY));
    
    // 发送起始信号
    I2C_GenerateSTART(COMPASS_I2C, ENABLE);
    while (!I2C_CheckEvent(COMPASS_I2C, I2C_EVENT_MASTER_MODE_SELECT));
    
    // 发送设备地址 (写)
    I2C_Send7bitAddress(COMPASS_I2C, 罗盘旧方案_ADDR << 1, I2C_Direction_Transmitter);
    while (!I2C_CheckEvent(COMPASS_I2C, I2C_EVENT_MASTER_TRANSMITTER_MODE_SELECTED));
    
    // 发送寄存器地址
    I2C_SendData(COMPASS_I2C, reg);
    while (!I2C_CheckEvent(COMPASS_I2C, I2C_EVENT_MASTER_BYTE_TRANSMITTED));
    
    // 重新起始
    I2C_GenerateSTART(COMPASS_I2C, ENABLE);
    while (!I2C_CheckEvent(COMPASS_I2C, I2C_EVENT_MASTER_MODE_SELECT));
    
    // 发送设备地址 (读)
    I2C_Send7bitAddress(COMPASS_I2C, 罗盘旧方案_ADDR << 1, I2C_Direction_Receiver);
    while (!I2C_CheckEvent(COMPASS_I2C, I2C_EVENT_MASTER_RECEIVER_MODE_SELECTED));
    
    // 禁用ACK
    I2C_AcknowledgeConfig(COMPASS_I2C, DISABLE);
    
    // 读取数据
    while (!I2C_CheckEvent(COMPASS_I2C, I2C_EVENT_MASTER_BYTE_RECEIVED));
    val = I2C_ReceiveData(COMPASS_I2C);
    
    // 发送停止信号
    I2C_GenerateSTOP(COMPASS_I2C, ENABLE);
    
    // 重新使能ACK
    I2C_AcknowledgeConfig(COMPASS_I2C, ENABLE);
    
    return val;
}

/* ---- 初始化罗盘 ---- */
void compass_init(void)
{
    // 注意：I2C3已经在oled_init()中初始化，这里只需配置罗盘旧方案
    
    Delay_Ms(10);
    
    // SET/RESET周期
    compass_write_reg(罗盘旧方案_REG_SET_RESET, 0x01);
    Delay_Ms(10);
    
    // 配置: 200Hz ODR, ±2G, 512 OSR, 连续模式
    compass_write_reg(罗盘旧方案_REG_CONTROL1, 0x1D);
    Delay_Ms(10);
    
    // 读取芯片ID验证
    uint8_t chip_id = compass_read_reg(罗盘旧方案_REG_CHIP_ID);
    printf("罗盘旧方案 Chip ID: 0x%02X (expected 0xFF)\r\n", chip_id);
}

/* ---- 读取罗盘数据 ---- */
uint8_t compass_read_data(compass_data_t *data)
{
    uint8_t buf[6];
    
    // 读取6字节磁场数据
    for (uint8_t i = 0; i < 6; i++) {
        buf[i] = compass_read_reg(罗盘旧方案_REG_DATA_X_L + i);
    }
    
    // 组合数据
    data->mag_x = (int16_t)(buf[1] << 8 | buf[0]);
    data->mag_y = (int16_t)(buf[3] << 8 | buf[2]);
    data->mag_z = (int16_t)(buf[5] << 8 | buf[4]);
    
    // 读取状态
    data->status = compass_read_reg(罗盘旧方案_REG_STATUS);
    
    return (data->status & 0x01);  // 返回数据就绪标志
}

/* ---- 计算航向角 ---- */
float compass_get_heading(int16_t mag_x, int16_t mag_y, float declination)
{
    float heading = atan2f((float)mag_y, (float)mag_x) * 180.0f / 3.14159265f;
    heading += declination;
    
    if (heading < 0) heading += 360.0f;
    if (heading >= 360.0f) heading -= 360.0f;
    
    return heading;
}
```
